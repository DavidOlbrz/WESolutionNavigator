Sticky Header







        
<!DOCTYPE html>
<html lang="de">

    <meta charset="UTF-8">
    <title>Sticky Header</title>
    <style>h1 {
        border: black solid 1px;
        padding: 10px;
        position: sticky;
        top: 5px;
        background: rgba(255, 255, 255, 0.8);
    }
    </style>
</head>

Bild verstecken








        
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Bild verstecken</title>
    <style>
        div {
            border: black solid 1px;
            padding: 5px;
            width: fit-content;
            height: fit-content;
        }

        img {
            opacity: 0;
            width: 100%;
            height: 100%;
        }

        #hide:checked ~ div img {
            opacity: 100%;
        }
    </style>
</head>
<body>
<h1>Übung 2.2</h1>
<input id="hide" name="hide" type="checkbox" checked>
<label for="hide">hide and show via checkbox</label>
<br>

</body>
</html>
